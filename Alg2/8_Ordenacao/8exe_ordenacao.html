Algoritmos II - Algoritmos de Ordenação

1. Ordenação com Bubble Sort

algoritmo "Ordenar_Vetor_BubbleSort"
var
    vetor : vetor[1..10] de inteiro
    i, j, temp : inteiro

inicio
    para i de 1 ate 10 faca
        escreva("Digite o número na posição ", i, ": ")
        leia(vetor[i])
    fimpara

    para i de 1 ate 9 faca
        para j de 1 ate 10 - i faca
            se vetor[j] > vetor[j+1] entao
                temp <- vetor[j]
                vetor[j] <- vetor[j+1]
                vetor[j+1] <- temp
            fimse
        fimpara
    fimpara

    escreval("Vetor ordenado em ordem crescente:")
    para i de 1 ate 10 faca
        escreval(vetor[i])
    fimpara
fimalgoritmo

2. Ordenação com Insertion Sort

algoritmo "Ordenar_Vetor_InsertionSort"
var
    vetor : vetor[1..10] de inteiro
    i, j, chave : inteiro

inicio
    para i de 1 ate 10 faca
        escreva("Digite o número na posição ", i, ": ")
        leia(vetor[i])
    fimpara

    para i de 2 ate 10 faca
        chave <- vetor[i]
        j <- i - 1
        enquanto (j > 0) e (vetor[j] > chave) faca
            vetor[j + 1] <- vetor[j]
            j <- j - 1
        fimenquanto
        vetor[j + 1] <- chave
    fimpara

    escreval("Vetor ordenado em ordem crescente:")
    para i de 1 ate 10 faca
        escreval(vetor[i])
    fimpara
fimalgoritmo

3. Ordenação com Selection Sort

algoritmo "Ordenar_Vetor_SelectionSort"
var
    vetor : vetor[1..10] de inteiro
    i, j, min, temp : inteiro

inicio
    para i de 1 ate 10 faca
        escreva("Digite o número na posição ", i, ": ")
        leia(vetor[i])
    fimpara

    para i de 1 ate 9 faca
        min <- i
        para j de i + 1 ate 10 faca
            se vetor[j] < vetor[min] entao
                min <- j
            fimse
        fimpara

        se min <> i entao
            temp <- vetor[i]
            vetor[i] <- vetor[min]
            vetor[min] <- temp
        fimse
    fimpara

    escreval("Vetor ordenado em ordem crescente:")
    para i de 1 ate 10 faca
        escreval(vetor[i])
    fimpara
fimalgoritmo
