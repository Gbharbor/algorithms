Exercício 1: Algoritmo de Registro de Pessoa

algoritmo "Registro de Pessoa"
tipo Pessoa = registro
    nome : caractere
    idade : inteiro
    altura : real
    doadorOrgaos : logico
fimregistro

var
    cidadao : Pessoa

inicio
    escreva("Digite o nome da pessoa: ")
    leia(cidadao.nome)
    escreva("Digite a idade da pessoa: ")
    leia(cidadao.idade)
    escreva("Digite a altura da pessoa: ")
    leia(cidadao.altura)
    escreva("É doador de órgãos? (verdadeiro/falso): ")
    leia(cidadao.doadorOrgaos)

    escreval("Nome: ", cidadao.nome)
    escreval("Idade: ", cidadao.idade)
    escreval("Altura: ", cidadao.altura)
    escreval("Doador de Órgãos: ", cidadao.doadorOrgaos)
fimalgoritmo

Exercício 3: Algoritmo de Dados do Aluno com Vetor

algoritmo "Dados do Aluno com Vetor"
tipo Aluno = registro
    nome : caractere
fimregistro

var
    aluno : Aluno
    notas : vetor[1..4] de real
    i : inteiro
inicio
    escreva("Digite o nome do aluno: ")
    leia(aluno.nome)

    para i de 1 ate 4 faca
        escreva("Digite a nota do ", i, "º bimestre: ")
        leia(notas[i])
    fimpara

    escreval("\nDados do Aluno:")
    escreval("Nome: ", aluno.nome)
    para i de 1 ate 4 faca
        escreval("Nota do ", i, "º bimestre: ", notas[i])
    fimpara
fimalgoritmo

Exercício 4: Registro de Pedágios

algoritmo "Registro Pedágios"
tipo Registro_pedagios = registro
    nome: caractere
    quilometro: real
    cidade: caractere
    valor: real
fimregistro
var
    i: inteiro
    pedagios: vetor [1..10] de registro_pedagios
    valortotal: real
inicio
    para i de 1 ate 10 faca
        escreva("Digite o nome da rodovia: ")
        leia(pedagios[i].nome)
        escreva("Digite o Km da rodovia: ")
        leia(pedagios[i].quilometro)
        escreva("Digite a cidade: ")
        leia(pedagios[i].cidade)
        escreva("Valor do Pedágio: ")
        leia(pedagios[i].valor)
    fimpara
    valortotal <- 0.0
    para i de 1 ate 10 faca
        valortotal <- valortotal + pedagios[i].valor
    fimpara
    escreval("Valor total dos pedágios: ", valortotal)
fimalgoritmo

Exercício 6: Inverter Valores

        algoritmo "Inverter Valores"
        var
            a, b: inteiro
        procedimento trocarvalores(var x: inteiro, var y: inteiro)
        var
            temp: inteiro
        inicio
            temp <- x
            x <- y
            y <- temp
        fimprocedimento
        inicio
            escreva("Digite o primeiro valor (a): ")
            leia(a)
            escreva("Digite o segundo valor (b): ")
            leia(b)
            escreval("Valores antes da troca: a = ", a, ", b = ", b)
            trocarvalores(a, b)
            escreval("Valores após a troca: a = ", a, ", b = ", b)
        fimalgoritmo
