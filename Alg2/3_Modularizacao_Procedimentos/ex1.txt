Algoritmo "Calculadora_Multimodular"
    procedimento ADICAO(A: inteiro, B: inteiro)
    var
        resultado: inteiro
    inicio
        resultado <- A + B
        escreval("Resultado da adição: ", resultado)
    fim

    procedimento SUBTRACAO(A: inteiro, B: inteiro)
    var
        resultado: inteiro
    inicio
        resultado <- A - B
        escreval("Resultado da subtração: ", resultado)
    fim

    procedimento MULTIPLICACAO(A: inteiro, B: inteiro)
    var
        resultado: inteiro
    inicio
        resultado <- A * B
        escreval("Resultado da multiplicação: ", resultado)
    fim

    procedimento DIVISAO(A: inteiro, B: inteiro)
    var
        resultado: real
    inicio
        se B <> 0 entao
            resultado <- real(A) / B
            escreval("Resultado da divisão: ", resultado)
        senao
            escreval("Erro: Divisão por zero!")
        fimse
    fim

    // Módulo Principal
    var
        num1, num2, opcao: inteiro
    inicio
        escreval("Digite o primeiro número: ")
        leia(num1)
        escreval("Digite o segundo número: ")
        leia(num2)

        escreval("Escolha a operação desejada:")
        escreval("1 - Adição")
        escreval("2 - Subtração")
        escreval("3 - Multiplicação")
        escreval("4 - Divisão")
        leia(opcao)

        escolha opcao
            caso 1
                ADICAO(num1, num2)
            caso 2
                SUBTRACAO(num1, num2)
            caso 3
                MULTIPLICACAO(num1, num2)
            caso 4
                DIVISAO(num1, num2)
            outrocaso

        fimescolha
