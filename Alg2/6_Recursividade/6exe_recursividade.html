Algoritmos II - Recursividade

1. Fatorial Iterativo

algoritmo "Calcular_Fatorial"
var
    numero, resultado : inteiro

funcao Fatorial(n: inteiro): inteiro
var
    i, fat : inteiro
inicio
    fat <- 1
    para i de 1 ate n faca
        fat <- fat * i
    fimpara
    retorne fat
fimfuncao

2. Fatorial Recursivo

algoritmo "Calcular_Fatorial_Recursivo"
var
    numero, resultado : inteiro

funcao Fatorial(n: inteiro): inteiro
inicio
    se n = 0 entao
        retorne 1
    senao
        retorne n * Fatorial(n - 1)
    fimse
fimfuncao

3. Fibonacci Iterativo

algoritmo "Fibonacci Iterativo"
var
    numero, resultado: inteiro

funcao fibonacci(n: inteiro): inteiro
    var
        a, b, temp, i: inteiro
    a <- 0
    b <- 1

    se n = 0 entao
        retorne a
    fimse

    para i de 2 ate n faca
        temp <- a + b
        a <- b
        b <- temp
    fimpara

    retorne b
fimfuncao

4. Fibonacci Recursivo

algoritmo "Fibonacci Recursivo"
var
    numero, resultado: inteiro

funcao fibonacci(n: inteiro): inteiro
    se n = 0 entao
        retorne 0
    senao
        se n = 1 entao
            retorne 1
        senao
            retorne fibonacci(n - 1) + fibonacci(n - 2)
        fimse
    fimse
fimfuncao
