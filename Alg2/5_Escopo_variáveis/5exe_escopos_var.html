Algoritmos II - exe Escopos e variaveis

1. Troca Por Valor

algoritmo "Troca_Por_Valor"
var
    num1, num2 : inteiro

procedimento Troca(x: inteiro, y: inteiro)
var
    temp : inteiro
inicio
    temp <- x
    x <- y
    y <- temp
    escreval("Dentro do procedimento TROCA: x = ", x, " e y = ", y)
fimprocedimento

inicio
    escreva("Digite o primeiro valor (num1): ")
    leia(num1)
    escreva("Digite o segundo valor (num2): ")
    leia(num2)
    escreval("Valores antes da chamada: num1 = ", num1, " e num2 = ", num2)
    Troca(num1, num2)
    escreval("Valores após a chamada: num1 = ", num1, " e num2 = ", num2)
fimalgoritmo

4. Exemplo com Parâmetros no Procedimento

algoritmo "Exemplo_Parametros"
var
    total: real // Variável global para armazenar o total inicial
inicio
    // Inicializa o valor total
    total <- 1000 

    // Chama o procedimento ADICIONA_DESCONTO com um desconto de 50
    ADICIONA_DESCONTO(50, total)

    // Chama o procedimento ADICIONA_DESCONTO novamente com um desconto de 30
    ADICIONA_DESCONTO(30, total)

    // Exibe o valor de total após todas as operações
    escreva("Total final (não alterado pelo procedimento): ", total)
fimalgoritmo

procedimento ADICIONA_DESCONTO(valor: real, total: real)
inicio
    total <- total - valor
    escreva("Total com desconto (local): ", total)
fimalgoritmo
